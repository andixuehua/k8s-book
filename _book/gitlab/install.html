       --hostname 192.168.146.99  \
       -p 10022:22 -p 80:80 -p 443:443 \
       --restart always \
       --volume /data/gitlab/config:/etc/gitlab \
       --volume /data/gitlab/logs:/var/log/gitlab \
       --volume /data/gitlab/data:/var/opt/gitlab \
       --name gitlab \
       gitlab/gitlab-ce
